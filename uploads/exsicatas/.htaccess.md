# Proteger pasta de uploads
Options -Indexes

# Bloquear execução de PHP
<FilesMatch "\.(php|php3|php4|php5|phtml|pl|cgi)$">
    Order Deny,Allow
    Deny from all
</FilesMatch>

# Permitir acesso apenas a imagens
<FilesMatch "\.(jpg|jpeg|png|gif|webp)$">
    Order Allow,Deny
    Allow from all
</FilesMatch>
